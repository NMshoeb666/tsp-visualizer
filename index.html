Greedy Algorithm

// Nearest Neighbor Heuristic
FUNC GreedyTSP(cities):
path = [City A]
visited = {City A}
currentCity = City A
WHILE visited.count < totalCities:
minDist = Infinity
nearest = null
FOR each city in cities:
IF city NOT in visited:
dist = distance(currentCity, city)
IF dist < minDist:
minDist = dist
nearest = city
path.add(nearest)
visited.add(nearest)
currentCity = nearest
path.add(City A) // Return to start
RETURN path
